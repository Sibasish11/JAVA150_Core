# Write a program that implements a simple ReadWriteLock to allow multiple readers but only one writer at a time. [Very High]

```java

class SimpleReadWriteLock {
    private int readers = 0;       // number of active readers
    private int writers = 0;       // number of active writers
    private int writeRequests = 0; // number of writers waiting

    // Acquire read lock
    public synchronized void lockRead() throws InterruptedException {
        while (writers > 0 || writeRequests > 0) { 
            wait(); // wait if a writer is active or waiting
        }
        readers++;
    }

    // Release read lock
    public synchronized void unlockRead() {
        readers--;
        notifyAll(); // notify writers if no readers
    }

    // Acquire write lock
    public synchronized void lockWrite() throws InterruptedException {
        writeRequests++;
        while (readers > 0 || writers > 0) { 
            wait(); // wait if readers or writers active
        }
        writeRequests--;
        writers++;
    }

    public synchronized void unlockWrite() {
        writers--;
        notifyAll(); // notify readers/writers
    }
}

// Shared resource (like a database or file)
class SharedResource {
    private int data = 0;
    private final SimpleReadWriteLock lock = new SimpleReadWriteLock();

    public void read(String readerName) {
        try {
            lock.lockRead();
            System.out.println(readerName + " reading data: " + data);
            Thread.sleep(500); // simulate read time
            lock.unlockRead();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }

    public void write(String writerName, int value) {
        try {
            lock.lockWrite();
            System.out.println(writerName + " writing data: " + value);
            data = value;
            Thread.sleep(800); // simulate write time
            lock.unlockWrite();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }
    }
}

// Demo with multiple readers and writers
public class ReadWriteLockDemo {
    public static void main(String[] args) {
        SharedResource resource = new SharedResource();

        // Readers
        for (int i = 1; i <= 3; i++) {
            final String readerName = "Reader-" + i;
            new Thread(() -> {
                for (int j = 0; j < 3; j++) {
                    resource.read(readerName);
                }
            }).start();
        }

        // Writers
        for (int i = 1; i <= 2; i++) {
            final String writerName = "Writer-" + i;
            final int value = i * 10;
            new Thread(() -> {
                for (int j = 0; j < 2; j++) {
                    resource.write(writerName, value + j);
                }
            }).start();
        }
    }
}


```
