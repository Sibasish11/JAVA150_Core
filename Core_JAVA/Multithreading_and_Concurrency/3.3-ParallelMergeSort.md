# Implement a program to sort an array of integers using a parallel merge sort approach (divide array, sort sub-arrays in separate threads, then merge). [Very High]

```java

import java.util.Arrays;

class ParallelMergeSortTask extends Thread {
    private int[] arr;

    public ParallelMergeSortTask(int[] arr) {
        this.arr = arr;
    }

    @Override
    public void run() {
        parallelMergeSort(arr);
    }

    // Recursive parallel merge sort
    public static void parallelMergeSort(int[] arr) {
        if (arr.length <= 1) {
            return; // base case
        }

        int mid = arr.length / 2;
        int[] left = Arrays.copyOfRange(arr, 0, mid);
        int[] right = Arrays.copyOfRange(arr, mid, arr.length);

        // Create threads for left and right sorting
        ParallelMergeSortTask leftTask = new ParallelMergeSortTask(left);
        ParallelMergeSortTask rightTask = new ParallelMergeSortTask(right);

        leftTask.start();
        rightTask.start();

        try {
            leftTask.join();
            rightTask.join();
        } catch (InterruptedException e) {
            e.printStackTrace();
        }

        // Merge sorted halves back into original array
        merge(arr, left, right);
    }

    // Merge function
    private static void merge(int[] arr, int[] left, int[] right) {
        int i = 0, j = 0, k = 0;

        while (i < left.length && j < right.length) {
            if (left[i] <= right[j]) {
                arr[k++] = left[i++];
            } else {
                arr[k++] = right[j++];
            }
        }

        while (i < left.length) {
            arr[k++] = left[i++];
        }

        while (j < right.length) {
            arr[k++] = right[j++];
        }
    }
}

public class ParallelMergeSort {
    public static void main(String[] args) {
        int[] arr = { 38, 27, 43, 3, 9, 82, 10 };

        System.out.println("Original array: " + Arrays.toString(arr));

        ParallelMergeSortTask.parallelMergeSort(arr);

        System.out.println("Sorted array:   " + Arrays.toString(arr));
    }
}


```
